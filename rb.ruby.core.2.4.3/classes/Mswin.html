def ln_safe(src, dest, *opt)
  cmd = ["mklink", dest.tr("/", "\\"), src.tr("/", "\\")]
  cmd[1, 0] = opt
  return if system("cmd", "/c", *cmd)
  # TODO: use RUNAS or something
  puts cmd.join(" ")
end
