static VALUE
stat_for_fd(VALUE self, VALUE fileno)
{
    struct stat st;
    if (fstat(NUM2INT(fileno), &st)) rb_sys_fail(0);
    return rb_stat_new(&st);
}

static VALUE
stat_for_path(VALUE self, VALUE path)
{
    struct stat st;
    FilePathValue(path);
    if (stat(RSTRING_PTR(path), &st)) rb_sys_fail(0);
    return rb_stat_new(&st);
}
