def _show(dest, ary, syms)
  if Symbol === (sym = ary[0])
    unless syms.include?(sym)
      return
    end
  end
  ary.each do |e|
    case e
    when Array
      _show(dest, e, syms)
    when String
      dest.print e
    end
  end
end

def endif(sym)
  if @kwdonly
    @stdout.puts "#endif /* #{sym} */"
    return
  end
  unless sym == @top[0]
    restore
    raise ArgumentError, "#{sym} unmatch to #{@top[0]}"
  end
  @top = @stack.pop
end

def ifdef(sym)
  if @kwdonly
    @stdout.puts "#ifdef #{sym}"
    return
  end
  @stack << @top
  @top << tmp = [sym]
  @top = tmp
end
