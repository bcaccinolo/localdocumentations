def vcross(b)
  Vec.new(@y * b.z - @z * b.y,
          @z * b.x - @x * b.z,
          @x * b.y - @y * b.x)
end

def vnormalize
  len = vlength
  v = Vec.new(@x, @y, @z)
  if len > 1.0e-17 then
    v.x = v.x / len
    v.y = v.y / len
    v.z = v.z / len
  end
  v
end
