def self.download(name, *rest)
  if https?
    super("https://raw.githubusercontent.com/gcc-mirror/gcc/master/#{name}", name, *rest)
  else
    super("https://repo.or.cz/official-gcc.git/blob_plain/HEAD:/#{name}", name, *rest)
  end
end
